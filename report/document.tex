	% ------------------ بخش دوم: درونیابی ------------------
	\section{بخش اول: درونیابی \lr{(Interpolation)}}
	در این بخش، هدف یافتن تابعی پیوسته است که به دقت از نقاط تجربی زیر عبور کند:
	\[ \{(1,1), (2,3), (3,5), (4,8), (5,5), (6,2)\} \]
	این کار به دو روش چندجمله‌ای لاگرانژ (مرتبه بالا) و اسپلاین مکعبی طبیعی \lr{(Natural Cubic Spline)} انجام شد.
	
	\subsection{ضوابط ریاضی استخراج‌شده}
	با استفاده از محاسبات نمادین \lr{(Symbolic Computation)} در پایتون، ضابطه دقیق چندجمله‌ای لاگرانژ مرتبه ۵ به صورت زیر حاصل شد:
	\begin{equation}
		P(x) = 0.175x^5 - 2.9583x^4 + 18.375x^3 - 52.0417x^2 + 68.45x - 31.0
	\end{equation}
	
	برای روش اسپلاین مکعبی، تابع به صورت تکه‌ای در ۵ بازه به فرم زیر تعریف می‌شود که محاسبات پایتون مقادیر دقیق ضریب هر قطعه را بدین شکل استخراج کرد:
	\begin{equation}
		S(x) = \begin{cases} 
			0.6667(x-1)^3 - 2.0000(x-1)^2 + 3.3333(x-1) + 1.0 & x \in [1, 2] \\
			0.6667(x-2)^3 + 1.3333(x-2) + 3.0 & x \in [2, 3] \\
			-2.3333(x-3)^3 + 2.0000(x-3)^2 + 3.3333(x-3) + 5.0 & x \in [3, 4] \\
			1.6667(x-4)^3 - 5.0000(x-4)^2 + 0.3333(x-4) + 8.0 & x \in [4, 5] \\
			1.6667(x-5)^3 - 4.6667(x-5) + 5.0 & x \in [5, 6]
		\end{cases}
	\end{equation}
	
	\subsection{تحلیل نمودار درون‌یابی}
	\begin{figure}[H]
		\centering
		\includegraphics[width=0.85\textwidth]{interpolation_plot.png}
		\caption{نمودار درونیابی نقاط به روش لاگرانژ و اسپلاین مکعبی}
		\label{fig:interpolation}
	\end{figure}
	
	همان‌طور که در شکل \ref{fig:interpolation} دیده می‌شود، چندجمله‌ای لاگرانژ (خط‌چین آبی) به دلیل درجه‌ی بالا، در نزدیکی مرزها (به خصوص بازه ۵ تا ۶) دچار نوسان نسبتاً شدیدی شده است که نمونه‌ای از پدیده رونگ \lr{(Runge's Phenomenon)} است. در مقابل، اسپلاین مکعبی به دلیل تکه‌ای بودن و برقراری شرط پیوستگی مشتق اول و دوم در گره‌ها ($C^2\text{ continuity}$)، مسیر بسیار نرم‌تر، هموارتر و واقعی‌تری را از بین نقاط طی کرده است.

	\subsection{تحلیل رفتار الگوریتم‌ها}
	بر اساس شکل \ref{fig:charts} و شکل \ref{fig:boxplot} می‌توان رفتار الگوریتم‌ها را تحلیل کرد:
	\begin{itemize}
		\item \textbf{الگوریتم ۱ (Alg.1):} دارای زمان اجرای بسیار پایین و تقریباً ثابت است که نشان‌دهنده پیچیدگی زمانی از مرتبه $\mathcal{O}(1)$ یا $\mathcal{O}(\log n)$ است. نوسان بسیار ناچیز در نمودار جعبه‌ای، پایداری فوق‌العاده این الگوریتم را در داده‌های بزرگ تأیید می‌کند.
		\item \textbf{الگوریتم ۲ (Alg.2):} روند رشد زمان اجرای آن کاملاً خطی و با شیب ثابت ملایم است که پیچیدگی زمانی $\mathcal{O}(n)$ را تداعی می‌کند. این الگوریتم تعادلی منطقی بین کارایی و پایداری برقرار ساخته است.
		\item \textbf{الگوریتم ۳ (Alg.3):} رفتاری به شدت نوسانی و صعودی دارد. زمان اجرای آن در داده‌های کوچک معقول است اما با افزایش حجم داده به صورت چندجمله‌ای با درجه بالا یا نمایی رشد کرده و در داده‌ی ۶۰۰ کیلوبایت به زمان ۶۰۰ میلی‌ثانیه می‌رسد که نشان‌دهنده پیچیدگی ناپایدار زمانی (نظیر $\mathcal{O}(n^2)$) است.
	\end{itemize}

	\subsection{تحلیل علمی نتایج انتگرال‌گیری}
	با تحلیل جدول \ref{tab:integration_results} مشخص است که:
	\begin{enumerate}
		\item \textbf{روش سیمپسون ۱/۳} با ۱۰۰ بخش‌بندی، دارای کمترین خطا ($3.01 \times 10^{-9}$) بوده و بیشترین همگرایی با مقدار دقیق را دارد. علت آن درجه صلبیت و دقت بالاتر این روش نسبت به ذوزنقه‌ای در تقریب تابع صعودی نمایی به وسیله سهمی‌ها است.
		\item \textbf{روش ترکیب گوسی (Gauss-Legendre)} تنها با \textbf{۵ نقطه ارزیابی} توانسته است به خطای خیره‌کننده $7.78 \times 10^{-8}$ برسد. این نشان‌دهنده کارایی فوق‌العاده این روش است که بدون نیاز به تقسیم‌بندی‌های ریز و با کمترین حجم محاسبات، پاسخی بسیار دقیق ارائه می‌دهد.
		\item \textbf{روش ذوزنقه‌ای} ضعیف‌ترین عملکرد را دارد؛ زیرا این روش رفتار نمایی و منحنی تابع را در هر زیربازه به صورت خط راست فرض می‌کند که منجر به خطای برش بیشتری می‌شود.
	\end{enumerate}
